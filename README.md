## Genetic Algorithm

```rs
#[tokio::main]
async fn main() {
    let mut pool = GeneticField::new(
        // ENFJ
        MbtiFitness::new(1.0, 1.0, 1.0, 1.0),
        GeneticFieldStrategy::Tangent,
        0.01,
        100,
    );

    for i in 0..100 {
        println!("Loop {}", i + 1);
        pool.evolution().await;
    }
}
```
